cmake_minimum_required(VERSION 3.16)

project(MultiBlockBPE
    LANGUAGES C CXX CUDA
)

# ==== C++ Standard ====
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ==== CUDA Settings ====
set(CMAKE_CUDA_STANDARD 20)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# Allow separable compilation for CUDA
set(CMAKE_CUDA_SEPARABLE_COMPILATION ON)

# ==== Flex ====
find_package(FLEX REQUIRED)

FLEX_TARGET(
    MyLexer
    ${CMAKE_SOURCE_DIR}/src/lexer.l
    ${CMAKE_BINARY_DIR}/lexer.c
)

# ==== Include Directories ====
include_directories(${CMAKE_SOURCE_DIR}/include)

# ==== Gather Sources ====
file(GLOB CPP_SOURCES ${CMAKE_SOURCE_DIR}/src/*.cpp)
file(GLOB CU_SOURCES  ${CMAKE_SOURCE_DIR}/src/*.cu)   # CUDA sources

# ==== Executable ====
add_executable(MultiBlockBPE
    ${CPP_SOURCES}
    ${CU_SOURCES}
    ${FLEX_MyLexer_OUTPUTS}   # lexer.c generated by Flex
)
